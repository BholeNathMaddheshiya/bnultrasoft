Q1) Write a program in 'C' to print total numbers of odd integers and their sum in an array that contains 15 element.


Answer:

        
# include < stdio.h >
    int main(){
        int n=15,a[n],s=0,t=0;
        for(int i=0;i < n;i++){
        printf("Enter the element of index %d: ",i);
        scanf("%d",&a[i]);
        if(a[i]%2!=0){
            s=s+a[i];
            t=t+1;
        }
        }
        printf("Sum of all odd integer in array is: %d",s);
        printf("\nTotal number of odd integer is: %d",t);
      return 0;
    }

Q3) Write a program in 'C' to take an integer value and return the number with its digits reversed.


Answer:

        
#include < stdio.h >
    int main(){
        int num, n, rev;
        printf("Enter any number: ");
        scanf("%d",&num);
        while(num>0){
        n=num%10;
        rev=rev*10+n;
        num=num/10;
        }
        printf("Reverse number is: %d",rev);
      return 0;
    }

Q5) Write a program in 'C' to convert decimal number into binary number. Example--Number 70 in decimal expression as 1000110 in binary.


Answer:

        
#include < stdio.h >
    int main(){
        int num,n, b=0,t=1;
        printf("Enter any number: ");
        scanf("%d",&num);
        while(num>0){
        n=num%2;
        b=n*t+b;
        t=t*10;
        num=num/2;
        }
        printf("Binary number is: %d",b);
      return 0;
    }
